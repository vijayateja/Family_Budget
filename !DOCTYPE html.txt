    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, onAuthStateChanged, signInWithCustomToken, signInAnonymously } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, doc, onSnapshot, collection, addDoc, deleteDoc, updateDoc, setDoc, getDocs, query, where } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        // Global State
        const today = new Date();
        window.user = null;
        window.revenueEntries = [];
        window.creditTransactions = [];
        window.fixedExpenses = [];
        window.currentViewMonth = today.getMonth();
        window.currentViewYear = today.getFullYear();
        window.localExpenseChanges = {};
        window.dashChart = null;
        
        window.creditLimits = { venture: 1000, savor: 400, debit: 0, megha: 0 };

        const BASELINE_COSTS = [
            { name: "Housing", amount: 2200 },
            { name: "Groceries", amount: 1000 },
            { name: "Auto", amount: 460 },
            { name: "Gas", amount: 450 },
            { name: "Utilities", amount: 200 },
            { name: "Mobile & Internet", amount: 160 },
            { name: "Insurance", amount: 160 }
        ];

        const firebaseConfig = JSON.parse(__firebase_config);
        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'portfolio-os-v2';

        window.updateFilteredViews = () => {
            renderRevenueLog();
            renderCreditLedger();
            renderExpenses();
            updateAll();
        };

        const renderRevenueLog = () => {
            const tbody = document.getElementById('revenue-log-body');
            if (!tbody) return;
            tbody.innerHTML = '';
            const filtered = window.revenueEntries.filter(e => {
                const d = new Date(e.date + 'T12:00:00');
                return d.getMonth() === window.currentViewMonth && d.getFullYear() === window.currentViewYear;
            });
            filtered.sort((a,b) => new Date(b.date) - new Date(a.date)).forEach(e => {
                const tr = document.createElement('tr');
                tr.innerHTML = `<td class="py-4 px-4 text-slate-500 font-mono text-[11px]">${e.date}</td><td class="py-4 px-4 text-white font-bold">${e.source}</td><td class="py-4 px-4 text-right text-emerald-400 font-bold">$${e.amount.toFixed(2)}</td><td class="py-4 px-4 text-right"><button onclick="window.removeRevenueFromCloud('${e.id}')" class="text-slate-800 hover:text-rose-500">✕</button></td>`;
                tbody.appendChild(tr);
            });
            document.getElementById('revenue-total-display').innerText = `$${filtered.reduce((s,e)=>s+e.amount,0).toLocaleString(undefined, {minimumFractionDigits: 2})}`;
        };

        const renderCreditLedger = () => {
            const tbody = document.getElementById('credit-ledger-body');
            if (!tbody) return;
            tbody.innerHTML = '';
            let running = { venture: 0, savor: 0, debit: 0, megha: 0 };
            const allSorted = [...(window.creditTransactions || [])].sort((a,b) => new Date(a.date) - new Date(b.date));
            const rows = allSorted.map(t => {
                const nodeKey = t.node.toLowerCase();
                if (running.hasOwnProperty(nodeKey)) running[nodeKey] += t.cost;
                const d = new Date(t.date + 'T12:00:00');
                return { ...t, currentRunning: running[nodeKey] || 0, viewMonth: d.getMonth(), viewYear: d.getFullYear() };
            });
            rows.filter(r => r.viewMonth === window.currentViewMonth && r.viewYear === window.currentViewYear)
                .reverse().forEach(t => {
                    const tr = document.createElement('tr');
                    const nodeLabel = t.node.toLowerCase();
                    const badgeClass = nodeLabel === 'megha' ? 'text-pink-400' : (nodeLabel === 'debit' ? 'text-emerald-400' : 'text-indigo-400');
                    tr.innerHTML = `
                        <td class="py-4 px-4 text-slate-500 font-mono text-[11px]">${t.date}</td>
                        <td class="py-4 px-4 text-white">${t.store} <span class="text-[8px] ${badgeClass} font-bold px-1 border border-white/10 rounded ml-2 uppercase">${t.node}</span></td>
                        <td class="py-4 px-4 text-right ${t.type === 'repay' ? 'text-emerald-400' : 'text-rose-400'} font-bold">$${Math.abs(t.cost).toFixed(2)}</td>
                        <td class="py-4 px-4 text-right text-slate-400 font-mono">$${t.currentRunning.toFixed(2)}</td>
                        <td class="py-4 px-4 text-right"><button onclick="window.removeCreditFromCloud('${t.id}')" class="text-slate-800 hover:text-rose-500">✕</button></td>
                    `;
                    tbody.appendChild(tr);
                });
        };

        const updateAll = () => {
            const currentExpenses = window.fixedExpenses.map(e => window.localExpenseChanges[e.id] || e);
            const totalMonthlyTarget = currentExpenses.reduce((s, e) => s + (parseFloat(e.amount) || 0), 0);
            
            const filteredLedger = window.creditTransactions.filter(e => {
                const d = new Date(e.date + 'T12:00:00');
                return d.getMonth() === window.currentViewMonth && d.getFullYear() === window.currentViewYear;
            });

            const actualSpentOnTarget = currentExpenses.reduce((s, exp) => {
                const spent = filteredLedger
                    .filter(t => t.store === exp.name && t.type === 'expense')
                    .reduce((total, t) => total + t.cost, 0);
                return s + spent;
            }, 0);

            const filteredRev = window.revenueEntries.filter(e => {
                const d = new Date(e.date + 'T12:00:00');
                return d.getMonth() === window.currentViewMonth && d.getFullYear() === window.currentViewYear;
            });
            const grossEarned = filteredRev.reduce((s, e) => s + e.amount, 0);

            const meghaThisPeriod = filteredLedger.filter(t => t.node.toLowerCase() === 'megha').reduce((s, t) => s + t.cost, 0);
            const debitThisPeriod = filteredLedger.filter(t => t.node.toLowerCase() === 'debit').reduce((s, t) => s + t.cost, 0);
            const ccRepayments = filteredLedger.filter(c => c.type === 'repay' && !['megha', 'debit'].includes(c.node.toLowerCase())).reduce((s,c) => s + Math.abs(c.cost), 0);

            document.getElementById('dashTotalIncome').innerText = `$${grossEarned.toLocaleString(undefined, {minimumFractionDigits: 2})}`; 
            document.getElementById('dashTotalExpenses').innerText = `$${totalMonthlyTarget.toLocaleString(undefined, {minimumFractionDigits: 2})}`; 
            const utilization = totalMonthlyTarget > 0 ? ((actualSpentOnTarget / totalMonthlyTarget) * 100).toFixed(0) : 0;
            document.getElementById('dashBudgetProgress').innerText = `UTILIZATION: ${utilization}% ($${actualSpentOnTarget.toLocaleString()})`;

            document.getElementById('dashMeghaTotal').innerText = `$${meghaThisPeriod.toLocaleString(undefined, {minimumFractionDigits: 2})}`;
            
            const netVal = grossEarned - ccRepayments - actualSpentOnTarget - debitThisPeriod;
            document.getElementById('dashNetFlow').innerText = `$${netVal.toLocaleString(undefined, {minimumFractionDigits: 2})}`;

            if (window.dashChart) {
                window.dashChart.data.labels = currentExpenses.map(e => e.name);
                window.dashChart.data.datasets[0].data = currentExpenses.map(e => e.amount);
                window.dashChart.update();
            }

            const summaryCont = document.getElementById('dashNodeSummary');
            let usage = { venture: 0, savor: 0, debit: 0, megha: 0 };
            filteredLedger.forEach(t => {
                const nodeKey = t.node.toLowerCase();
                if (usage.hasOwnProperty(nodeKey)) usage[nodeKey] += t.cost;
            });
            
            const displayNodes = Object.keys(window.creditLimits).filter(node => node !== 'megha');
            summaryCont.innerHTML = displayNodes.map(node => {
                const totalUsed = usage[node] || 0;
                const limit = window.creditLimits[node];
                if (limit === 0 && node === 'debit') {
                    return `<div class="p-4 bg-white/[0.02] rounded-xl border border-white/5 flex justify-between items-center group">
                        <span class="text-[10px] font-bold text-slate-500 uppercase">${node}</span>
                        <div class="text-right">
                             <span class="text-xs font-black text-white">$${totalUsed.toLocaleString(undefined, {minimumFractionDigits: 2})}</span>
                             <div class="text-[8px] text-slate-500 uppercase mt-0.5">Total Spent</div>
                        </div>
                    </div>`;
                }
                const rem = limit - totalUsed;
                const progressWidth = Math.max(0, Math.min(100, (totalUsed / limit) * 100));
                return `<div class="p-4 bg-white/[0.02] rounded-xl border border-white/5 space-y-3 group">
                    <div class="flex justify-between items-start">
                        <div class="cursor-pointer" onclick="window.triggerEditLimit('${node}')">
                            <span class="text-[10px] font-bold text-indigo-400 group-hover:text-indigo-300 transition-colors uppercase">${node} <span class="text-[8px] opacity-0 group-hover:opacity-100 ml-1">✎</span></span>
                            <div class="text-[8px] text-slate-600 uppercase mt-0.5">Total Limit: $${limit.toLocaleString()}</div>
                        </div>
                        <div class="flex flex-col items-end">
                            <span class="text-xs font-black ${rem < 50 ? 'text-rose-500' : 'text-emerald-400'}">$${rem.toLocaleString(undefined, {minimumFractionDigits: 2})}</span>
                            <span class="text-[8px] text-slate-500 uppercase mt-0.5">Remaining</span>
                        </div>
                    </div>
                    <div class="w-full bg-slate-800 h-1.5 rounded-full overflow-hidden">
                        <div class="h-full ${progressWidth > 90 ? 'bg-rose-500' : 'bg-indigo-500'}" style="width: ${progressWidth}%"></div>
                    </div>
                    <div class="flex justify-between items-center pt-1 border-t border-white/5">
                        <span class="text-[9px] font-mono text-slate-400">Used: <span class="text-white font-bold">$${totalUsed.toLocaleString(undefined, {minimumFractionDigits: 2})}</span></span>
                        <button onclick="window.triggerQuickRepay('${node}')" class="text-[8px] font-bold text-emerald-500 uppercase tracking-widest px-2 py-1 rounded bg-emerald-500/5 border border-emerald-500/10 hover:bg-emerald-500/10 transition-all">Settle</button>
                    </div>
                </div>`;
            }).join('');

            const storeDropdown = document.getElementById('credit-store');
            if (storeDropdown) {
                const currentVal = storeDropdown.value;
                storeDropdown.innerHTML = `<option value="">Select Category...</option>` + 
                    currentExpenses.map(e => `<option value="${e.name}">${e.name}</option>`).join('') +
                    `<option value="Other">Other Expense</option>`;
                storeDropdown.value = currentVal;
            }
        };

        window.renderExpenses = () => {
            const list = document.getElementById('fixedExpenseList');
            if (!list) return;
            list.innerHTML = '';
            const filteredLedger = window.creditTransactions.filter(e => {
                const d = new Date(e.date + 'T12:00:00');
                return d.getMonth() === window.currentViewMonth && d.getFullYear() === window.currentViewYear;
            });
            window.fixedExpenses.forEach((e, i) => {
                const item = window.localExpenseChanges[e.id] || e;
                const periodPaid = filteredLedger
                    .filter(t => t.store === item.name && t.type === 'expense')
                    .reduce((s, t) => s + t.cost, 0);
                const remaining = Math.max(0, item.amount - periodPaid);
                const progress = Math.min(100, (periodPaid / item.amount) * 100) || 0;
                const isComplete = remaining <= 0;
                const card = document.createElement('div');
                card.className = `p-6 rounded-2xl border transition-all cursor-pointer ${isComplete ? 'bg-emerald-500/10 border-emerald-500/30' : 'bg-white/[0.03] border-white/5 hover:border-white/20'}`;
                card.onclick = () => window.editExpenseItem(i);
                card.innerHTML = `
                    <div class="flex justify-between mb-4">
                        <div class="max-w-[120px]">
                            <span class="text-[10px] font-bold ${isComplete ? 'text-emerald-500' : 'text-slate-500'} uppercase block tracking-wider truncate">${item.name}</span>
                            <span class="text-[10px] text-slate-500 mt-1 block">Goal: <span class="text-white font-bold">$${item.amount.toLocaleString()}</span></span>
                        </div>
                        <div class="text-right">
                            <span class="text-[10px] font-bold uppercase block tracking-wider ${isComplete ? 'text-emerald-500' : 'text-slate-500'}">Balance Due</span>
                            <span class="text-xl font-black ${isComplete ? 'text-emerald-400' : 'text-rose-400'}">$${remaining.toLocaleString(undefined, {minimumFractionDigits: 2})}</span>
                        </div>
                    </div>
                    <div class="w-full h-1.5 bg-slate-800 rounded-full overflow-hidden mb-2">
                        <div class="h-full ${isComplete ? 'bg-emerald-500' : 'bg-indigo-500'} transition-all duration-500" style="width: ${progress}%"></div>
                    </div>
                    <div class="flex justify-between items-center text-[9px] font-mono">
                        <span class="${isComplete ? 'text-emerald-500' : 'text-slate-500'} uppercase font-bold">Logged: $${periodPaid.toLocaleString(undefined, {minimumFractionDigits: 2})}</span>
                    </div>
                `;
                list.appendChild(card);
            });
        };

        const setupDataListeners = async (uid) => {
            const expRef = collection(db, 'artifacts', appId, 'users', uid, 'fixed_expenses');
            const limitRef = collection(db, 'artifacts', appId, 'users', uid, 'credit_limits');
            const snapExp = await getDocs(expRef);
            if (snapExp.empty) { for (const cost of BASELINE_COSTS) { await addDoc(expRef, cost); } }
            const snapLimits = await getDocs(limitRef);
            if (snapLimits.empty) { await setDoc(doc(limitRef, 'default'), window.creditLimits); }
            onSnapshot(collection(db, 'artifacts', appId, 'users', uid, 'revenue'), (snap) => {
                window.revenueEntries = snap.docs.map(doc => ({ id: doc.id, ...doc.data() }));
                window.updateFilteredViews();
            });
            onSnapshot(collection(db, 'artifacts', appId, 'users', uid, 'credit_ledger'), (snap) => {
                window.creditTransactions = snap.docs.map(doc => ({ id: doc.id, ...doc.data() }));
                window.updateFilteredViews();
            });
            onSnapshot(collection(db, 'artifacts', appId, 'users', uid, 'fixed_expenses'), (snap) => {
                window.fixedExpenses = snap.docs.map(doc => ({ id: doc.id, ...doc.data() }));
                window.updateFilteredViews();
            });
            onSnapshot(doc(db, 'artifacts', appId, 'users', uid, 'credit_limits', 'default'), (snap) => {
                if (snap.exists()) { window.creditLimits = snap.data(); window.updateFilteredViews(); }
            });
        };

        window.addCreditToCloud = async (tx) => { if (window.user) await addDoc(collection(db, 'artifacts', appId, 'users', window.user.uid, 'credit_ledger'), tx); };
        window.removeCreditFromCloud = async (id) => { if (window.user) await deleteDoc(doc(db, 'artifacts', appId, 'users', window.user.uid, 'credit_ledger', id)); };
        window.updateExpenseInCloud = async (id, data) => { if (window.user) await updateDoc(doc(db, 'artifacts', appId, 'users', window.user.uid, 'fixed_expenses', id), data); };
        window.updateLimitsInCloud = async (data) => { if (window.user) await setDoc(doc(db, 'artifacts', appId, 'users', window.user.uid, 'credit_limits', 'default'), data); };

        window.handleAddRevenue = async () => {
            const amount = parseFloat(document.getElementById('entry-amount').value) || 0;
            if (amount <= 0) return;
            await window.addRevenueToCloud?.({ date: document.getElementById('entry-date').value, source: document.getElementById('entry-source').value, amount, timestamp: Date.now() });
            document.getElementById('entry-amount').value = '';
            showToast("Revenue Logged");
        };
        window.handleAddCredit = async () => {
            const cost = parseFloat(document.getElementById('credit-cost').value) || 0;
            const storeSelection = document.getElementById('credit-store').value;
            const type = document.getElementById('credit-type').value;
            if (cost <= 0 || !storeSelection) return;
            await window.addCreditToCloud?.({
                date: document.getElementById('credit-date').value, node: document.getElementById('credit-node').value,
                store: storeSelection, type: type, cost: cost * (type === 'repay' ? -1 : 1), timestamp: Date.now()
            });
            document.getElementById('credit-cost').value = '';
            document.getElementById('credit-store').value = '';
            showToast("Action Recorded");
        };
        window.triggerQuickRepay = (node) => {
            document.getElementById('modalTitle').innerText = `Settle ${node.toUpperCase()}`;
            document.getElementById('modalInputs').innerHTML = `<div class="space-y-4">
                <input type="date" id="repay-date" class="premium-input w-full" value="${new Date().toISOString().split('T')[0]}">

                if (amount <= 0) return;
                await window.addCreditToCloud?.({
                    date: document.getElementById('repay-date').value, node: node,
                    store: "Balance Repayment", type: 'repay', cost: -amount, timestamp: Date.now()
                });
                closeModal();
            };
            document.getElementById('aiModal').classList.remove('opacity-0', 'pointer-events-none');
        };

            document.getElementById('modalConfirmBtn').onclick = async () => {
                const newVal = parseFloat(document.getElementById('edit-limit-val').value) || 0;

                closeModal();
            };

        };

            };
            document.getElementById('aiModal').classList.remove('opacity-0', 'pointer-events-none');
        };
